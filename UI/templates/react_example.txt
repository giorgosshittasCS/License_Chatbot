import React, { useState, useEffect } from 'react';

const Chatbot = () => {
  const [question, setQuestion] = useState('');

  useEffect(() => {
    // Fetch initial question when component mounts
    fetchQuestion();
  }, []);

  const fetchQuestion = async () => {
    const response = await fetch('/question');
    const data = await response.json();
    setQuestion(data.question);
  };

  const handleAnswer = async (answer) => {
    // Handle answer logic and fetch the next question
    const response = await fetch('/question', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({ answer }),
    });

    const data = await response.json();
    setQuestion(data.question);
  };

  return (
    <div>
      <div className="chat-block">
        <div className="bot-icon-block">
          <img className="bot-image" src="/images/bot.png" alt="" />
        </div>
        <div className="question-block">{question}</div>
      </div>
      <div className="options-block">
        <button className="yes-block option" onClick={() => handleAnswer('Yes')}>
          Yes
        </button>
        <button className="no-block option" onClick={() => handleAnswer('No')}>
          No
        </button>
        <button className="neutral-block option" onClick={() => handleAnswer("Don't Mind")}>
          Don't Mind
        </button>
      </div>
    </div>
  );
};

export default Chatbot;
